  • String Methods
    •    

const learning = 'Learning JavaScript is great!';
let output;

// length Conta os caracteres
output = learning.length ;

// concat cada parametro concatena a string
output = learning.concat(" ", " and fun");

// uppercase Transforma em caixa alta / Letra maiuscula
output = learning.toUpperCase();

// lowercase Transforma em caixa Baix / Letra minuscula
output = learning.toLowerCase();

// indexOf // indexof -1 quer dizer que nao existe.
// No demais retorna onde esta a primeira letra da string dividindo ela num array
output = learning.indexOf('JavaScript');
output = learning.indexOf('PHP');

// substring() corta letras da palavra
// Primeiro parametro eh do inicio para frente
// Segundo parametro eh do fin para o inicio
output = learning.substring(0,10);
output = learning.substring(2,10);

// Aquie podemos mostrar apenas os 4 ultimos caracteres
output = learning.substring( learning.length - 4 );

// slice() corta do inicio para frente e com
// Numero negativo corta da frente para atras
// negative number will start from the end
output = learning.slice(0,4); //Lear
output = learning.slice(1,4); //ear
output = learning.slice(-2);  //t!

// Split retira o espaço da string
output = learning.split(" ");

// Outro exemplo retira a virgula e o espaço
const hobbies = 'read, walk, listen music, write, learn to program';
output = hobbies.split(", ");

// Replace
//Primeiro o valor novo depois o valor que sera removido
output = learning.replace('JavaScript', 'Modern JavaScript');

// includes
// Verifica se contem a string dentro da string e retorna true ou false
output = learning.includes('JavaScript');
output = learning.includes('PHP');

// repeat
let greet =  "hello ";
    output = greet.repeat(3);

console.log(output);
